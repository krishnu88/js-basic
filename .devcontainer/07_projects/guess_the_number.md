```javascript


//first we select or capture all values that are required.since number is obtained as string(or can be a string) so we apply parseInt(). 
const userInput=document.querySelector("#guessField")
//since we want to guess number between 1 and 100,so we generate random number only between 1 and 100.we also add 1 to avoid occurence of 0.parseInt() is used to get integer part of generated random number.
let randomNumber=parseInt(Math.random() * 100+1)
//console.log(randomNumber)
//now we select the submit button 
const submit=document.querySelector("#subt")
//now we select the numbers that are being already guessed.
let guessedSlot=document.querySelector(".Guesses")
//now we select the number of guesses left to guess the number
const guessesLeft=document.querySelector(".lastResult")
//now we select hiOrLow
const lowOrHi=document.querySelector(".lowOrHi")
//now we select the resultParas
const startOver=document.querySelector(".resultParas")
//now we create a new element of paragraph(p tag) using createElement()
const p = document.createElement("p")
//now we start by initialising our no. of guesses to 1. To display the guesses the user has alraedy made,we also create an empty array to display the previously guessed values.
let numberGuesses=1;
let previousGuesses=[] ;//creation of empty array.
//now we create a variable to tell that the user is available to play the game.
let userAvailabilty=true;
//now we create four functions: ValidityOfGuess(),checkGuess(),displayGuess(),displayMessage(),endGame(),startGame()
//now we check whether the user is elgible to play the game or not.
if(userAvailabilty)  //gets executed only if the userAvailabilty
{
    submit.addEventListener("click",function (e){         //now if userAvailabilty=true;then we add eventlistener click that selects the value from input field.
      e.preventDefault()  //this method prevents the default action of button to submit values.
      const guess = parseInt(userInput.value)
//now after all this,goto function validityOfGuess.
validityOfGuess(guess)
    });
}
//validityOfGuess()  =>  this function checks whether the value entered by user is appropriate or not.it takes
function validityOfGuess(guess)
{
  if(isNaN(guess))    //now we check whether the value obtained from input is correct or not.(that is can it be evaluated or not.)
  {
    alert("please enter a valid number.")
  }
  else if(guess > 100)
  {
    alert("please enter a valid number.")
  }
  else if(guess < 1)
  {
    alert("please enter a valid number.")
  }
  //if it is true then we move forward.
  else
  {
    previousGuesses.push(guess);    //we push the already guessed values inside array using push() so that we can display these values latter on. 
    if(numberGuesses===11)     //now we check number of attempts made to guess the number.if maximum limit is reached then we display game over message and move to endGame function.
    {
      displayGuessedValues(guess)
      displayMessages(`game over.Random number was ${randomNumber}`)
      endGame()
      
    }
    //if number of attempts to guess the number have not been reached yet,then we move to displayGuess() function.after that we move to checkGuess() function
    else
    {
      displayGuessedValues(guess)
      checkGuess(guess)
    }
  }
}
//now we make a function checkGuess to actually compare the random number generated by web with the number as given by user.here we make 3 cases.
function checkGuess(guess)
{
   if(guess===randomNumber)  //checks whether the randomNumber is equal to number given as input.
   {
      displayMessages(`you guessed it right!hurray,you won the game in ${guesses} number of attempts`)
      endGame()
   }
   else if(guess < randomNumber)    //otherwise we compare the values to give hints to user to guess the number.
        {
            displayMessages(`guessed number is TOOOO low `);
        }
        else if(guess > randomNumber)
        {
          displayMessages(`guessed number is TOOOO high `)
        }        
}
//now we move to displayGuessdValues function.this function sets value of variable to default values so that we can enter the next guessed value.it also displays the already guessed values.
function displayGuessedValues(guess)
{
   //this clears the value of input given by user.
  //this again and again adds values to the empty array guessedNumslot.
  
  // guessedSlot.innerHTML += `${guess}, `;
  userInput.value=""
  //guessedSlot.innerHTML += `${guess}`
numberGuesses++;
guessedSlot.innerHTML += `${guess} ; `
guessesLeft.innerHTML= `${11 - numberGuesses}`
}
//displayMessage()  =>  this function displays the message on the screen in the hiOrLow class field
function displayMessages(message)
{
lowOrHi.innerHTML=`<h2>${message}</h2>`

}
//so we now we move to endGame() function 
function endGame()
{//clears the value entered by user
//next we set our userinput to disable so that it does not accept any furthur values.
//now we create a button to end game and start game again.however there is no need to create a new button.we can also create text and add attribute to it so that is behaves like button.
//first we add text to p tag before coverting it to button
//using p tag as button.
//now we need to add paragraph (p tag) to the resultParas.this is done using appendChild().
//since we need to end game,so we set userAvailabilty to false.
userInput.value = '';
userInput.setAttribute('disabled', '');
p.classList.add('button');
p.innerHTML = `<h2 id="newGame">Start new Game</h2>`;
startOver.appendChild(p);
userAvailabilty = false;
newGame();
}
function newGame()
{      
  //first we select the start the game button which is nothing but our p tag which we created.this is done using querySelector() by specifying its id within "" inside ().
  const newGameButton = document.querySelector('#newGame');
  newGameButton.addEventListener('click', function (e) {
    randomNumber = parseInt(Math.random() * 100 + 1);
    previousGuesses = [];
    numberGuesses = 1;
    guessedSlot.innerHTML = '';
    guessesLeft.innerHTML = `${11 - numberGuesses} `;
    userInput.removeAttribute('disabled');
    startOver.removeChild(p);

   userAvailabilty= true;
  });
}
```
